bool runloop = true;
HHD device;
HDSchedulerHandle gSchedulerCallback = HD_INVALID_HANDLE;
// RedisClient redis_client;

void sighandler(int sig) { runloop = false; }
void exitHandler()
{
    hdStopScheduler();
    hdUnschedule(gSchedulerCallback);

    if (device != HD_INVALID_HANDLE)
    {
        hdDisableDevice(device);
        device = HD_INVALID_HANDLE;
    }
}

HDCallbackCode HDCALLBACK phantomCallback(void* data)
{
    HHD _device = hdGetCurrentDevice();
    hdBeginFrame(_device);

    hduVector3Dd pos, force;
    hdGetDoublev(HD_CURRENT_POSITION, pos);
    hdSetDoublev(HD_CURRENT_FORCE, force);

    std::cout << pos << std::endl;

    hdEndFrame(_device);

    HDErrorInfo error;
    if (HD_DEVICE_ERROR(error = hdGetError()))
    {
        std::cout << "Error during scheduling callback" << "Error: " << &error << std::endl;
        // if (hduIsSchedulerError(&error))
        // {
        //     return HD_CALLBACK_DONE;
        // }
    }
    return HD_CALLBACK_CONTINUE;
}

void runPhantomLoop()
{
    gSchedulerCallback = hdScheduleAsynchronous(phantomCallback,0,HD_DEFAULT_SCHEDULER_PRIORITY);

    HDErrorInfo error;
    if (HD_DEVICE_ERROR(error = hdGetError()))
    {
        std::cout << "failed to start scheduler. " << "Error: " << &error << std::endl;
        exit(-1);
    }

    while (runloop)
    {
        /* code */
    }
}


Eigen::Vector3d position;
Eigen::Vector3d sensed_force;


int main(int argc, char const *argv[])
{
    HDErrorInfo error;

    std::cout << "Opeing the device handler" << std::endl;
    atexit(exitHandler);

    // Initialize the device.
    device = hdInitDevice(HD_DEFAULT_DEVICE);
    if (HD_DEVICE_ERROR(error = hdGetError()))
    {
        // hduPrintError(stderr,&error, "failed to initialize device.");
        std::cout << "failed to initialize device." << "Error: " << &error << std::endl;
        exit(-1);
    }
    std::cout << "found device: " << hdGetString(HD_DEVICE_MODEL_TYPE) << std::endl;
    hdEnable(HD_FORCE_OUTPUT);
    hdEnable(HD_MAX_FORCE_CLAMPING);

    hdStartScheduler();
    if (HD_DEVICE_ERROR(error = hdGetError()))
    {
        std::cout << "failed to start scheduler. " << "Error: " << &error << std::endl;
        return -1;
    }
    // start the phantom loop.
    runPhantomLoop();

    return 0;
}
